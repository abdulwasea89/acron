  Premium real-time is invisible when it works and honest when it doesn't. Four
  layers:

  1. A rock-solid connection hook вҖ” auto-reconnect with backoff, heartbeat, and
  clean teardown. This is the foundation; everything else rides on it.

  // frontend/hooks/useRealtime.ts
  "use client";
  import { useEffect, useRef, useState } from "react";

  type RtStatus = "connecting" | "live" | "reconnecting" | "offline";
  type RtEvent = { type: string; [k: string]: unknown };

  export function useRealtime(onEvent: (e: RtEvent) => void) {
    const [status, setStatus] = useState<RtStatus>("connecting");
    const onEventRef = useRef(onEvent);
    onEventRef.current = onEvent;                    // always call latest 
  handler

    useEffect(() => {
      let ws: WebSocket | null = null;
      let heartbeat: ReturnType<typeof setInterval> | null = null;
      let retry: ReturnType<typeof setTimeout> | null = null;
      let attempt = 0;
      let closed = false;

      async function connect() {
        try {
          const res = await fetch("/api/ws-ticket");
          if (!res.ok) throw new Error("no ticket");
          const { token } = await res.json();
          const proto = location.protocol === "https:" ? "wss" : "ws";
          const base = process.env.NEXT_PUBLIC_WS_URL ??
  `${proto}://${location.host}`;
          ws = new
  WebSocket(`${base}/api/v1/ws?token=${encodeURIComponent(token)}`);

          ws.onopen = () => {
            attempt = 0;
            setStatus("live");
            heartbeat = setInterval(() => ws?.readyState === 1 &&
  ws.send("ping"), 25_000);
          };
          ws.onmessage = (m) => {
            const e = JSON.parse(m.data) as RtEvent;
            if (e.type === "pong" || e.type === "connected") return;
            onEventRef.current(e);
          };
          ws.onclose = () => {
            if (heartbeat) clearInterval(heartbeat);
            if (closed) return;
            setStatus("reconnecting");
            const delay = Math.min(1000 * 2 ** attempt++, 15_000);  // 1sвҶ’2sвҶ’4sвҖҰ
  cap 15s
            retry = setTimeout(connect, delay);
          };
          ws.onerror = () => ws?.close();
        } catch {
          setStatus("reconnecting");
          retry = setTimeout(connect, Math.min(1000 * 2 ** attempt++, 15_000));
        }
      }

      connect();
      const onOnline = () => { if (ws?.readyState !== 1) { attempt = 0;
  connect(); } };
      window.addEventListener("online", onOnline);
      window.addEventListener("offline", () => setStatus("offline"));

      return () => {
        closed = true;
        window.removeEventListener("online", onOnline);
        if (heartbeat) clearInterval(heartbeat);
        if (retry) clearTimeout(retry);
        ws?.close();
      };
    }, []);

    return status;
  }
